from itertools import combinations
def solution(nums):
    answer = 0
    answer_list = []
    for i in list(combinations(nums, 3)):
        answer_list.append(sum(i))
    # answer_list = list(set(answer_list))

    for num in answer_list:
        for j in range(2,num):
            if (num % j) == 0:
                break
        else:
            answer+=1

    return answer
